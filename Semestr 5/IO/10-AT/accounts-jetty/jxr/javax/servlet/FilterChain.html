28  package javax.servlet;

30  import java.io.IOException;

32      /**
33      * A FilterChain is an object provided by the servlet container to the developer
34      * giving a view into the invocation chain of a filtered request for a resource. Filters
35      * use the FilterChain to invoke the next filter in the chain, or if the calling filter
36      * is the last filter in the chain, to invoke the resource at the end of the chain.
37      *
38      * @see Filter
39      * @since Servlet 2.3
40      **/
41  
42  public interface FilterChain {
43  	
44  	/**
45  	* Causes the next filter in the chain to be invoked, or if the calling filter is the last filter
46  	* in the chain, causes the resource at the end of the chain to be invoked.
47  	*
48  	* @param request the request to pass along the chain.
49  	* @param response the response to pass along the chain.
50  	*
51  	* @since 2.3
52  	*/
53  	
54      public void doFilter ( ServletRequest request, ServletResponse response ) throws IOException, ServletException;
55  
56  }
