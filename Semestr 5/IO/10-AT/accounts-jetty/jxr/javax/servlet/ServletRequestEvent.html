28  package javax.servlet;
29  
30  
31      /** 
32        * Events of this kind indicate lifecycle
33        * events for a ServletRequest.
34        * The source of the event
35        * is the ServletContext of this web application.
36        * @see ServletRequestListener
37        * @since	Servlet 2.4
38        */
39  
40  public class ServletRequestEvent extends java.util.EventObject { 
41      private ServletRequest request;
42  
43      /** Construct a ServletRequestEvent for the given ServletContext
44        * and ServletRequest.
45        *
46        * @param sc		the ServletContext of the web application.
47        * @param request		the ServletRequest that is sending the event.
48        */
49      public ServletRequestEvent(ServletContext sc, ServletRequest request) {
50          super(sc);
51          this.request = request;
52      }
53      
54      /**
55        * Returns the ServletRequest that is changing.
56        */
57      public ServletRequest getServletRequest () { 
58          return this.request;
59      }
60  
61      /**
62        * Returns the ServletContext of this web application.
63        */
64      public ServletContext getServletContext () { 
65          return (ServletContext) super.getSource();
66      }
67  }
