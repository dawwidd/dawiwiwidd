28  package javax.servlet.http;
29  
30  import java.util.EventListener;
31  
32  	/** This listener interface can be implemented in order to
33  	* get notifications of changes to the attribute lists of sessions within
34  	* this web application.
35  	* @since	v 2.3
36  */
37  
38  public interface HttpSessionAttributeListener extends EventListener {
39  	/** Notification that an attribute has been added to a session. Called after the attribute is added.*/
40      public void attributeAdded ( HttpSessionBindingEvent se );
41  	/** Notification that an attribute has been removed from a session. Called after the attribute is removed. */
42      public void attributeRemoved ( HttpSessionBindingEvent se );
43  	/** Notification that an attribute has been replaced in a session. Called after the attribute is replaced. */
44      public void attributeReplaced ( HttpSessionBindingEvent se );
45  
46  }
